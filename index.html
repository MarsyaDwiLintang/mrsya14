class StringUtils {
  static String capitalize(String text) {
    if (text.isEmpty) return text;

  }

    return text.contains('@');
  }
}

  print('Capitalized name: $capitalized');
  print('Is "test@mail.com" an email? $isEmail');
}

05. Extension untuk Null Safety

Extension untuk Null Safety digunakan untuk membantu menangani data nullable seperti String? atau int? agar lebih aman saat digunakan. Dengan extension ini, programmer dapat menghindari error null dan membuat kode menjadi lebih singkat, rapi, serta mudah dibaca.

DART SOURCE CODE

// Nama  : Marsya Dwi Lintang
// NIM   : 251410028
// Kelas : SI2A

extension NullableStringExtensions on String? {
  // Safe access untuk nullable string
  String orEmpty() => this ?? '';

  bool get isNullOrEmpty =>
      this == null || this!.isEmpty;

  bool get isNotNullOrEmpty =>
      !isNullOrEmpty;

  String withDefault(String defaultValue) =>
      this ?? defaultValue;
}

void main() {
  // Penggunaan
  String? name;

  print(name.orEmpty());              // ''
  print(name.isNullOrEmpty);          // true
  print(name.isNotNullOrEmpty);       // false
  print(name.withDefault('Guest'));   // Guest
}

extension NumberExtensions on num {
  String toRupiah() {
    return 'Rp${toStringAsFixed(0).replaceAllMapped(
      RegExp(r'(\d{1,3})(?=(\d{3})+(?!\d))'),
      (match) => '${match[1]}.',
    )}';
  }

  bool isBetween(num min, num max) {
    return this >= min && this <= max;
  }

  num power(int exponent) {
    var result = this;

    for (var i = 1; i < exponent; i++) {
      result *= this;
    }

    return result;
  }
}

void main() {
  var price = 1500000;
  print(price.toRupiah());        // Rp1.500.000
  print(5.isBetween(1, 10));      // true
  print(2.power(3));              // 8
}

07. Generic Extensions

Generic Extensions digunakan agar extension dapat bekerja pada berbagai tipe data tanpa perlu membuat extension baru untuk setiap tipe. Dengan generic, extension menjadi lebih fleksibel, reusable, dan dapat digunakan pada banyak jenis data seperti List, List, dan lainnya.

DART SOURCE CODE

// Nama  : Marsya Dwi Lintang
// NIM   : 251410028
// Kelas : SI2A

extension ListExtensions on List {
  T? getOrNull(int index) {
    if (index < 0 || index >= length) {
      return null;
    }

    return this[index];
  }

  List> chunk(int size) {
    var chunks = >[];

    for (var i = 0; i < length; i += size) {
      chunks.add(
        sublist(
          i,
          i + size > length
              ? length
              : i + size,
        ),
      );
    }

    return chunks;
  }

  List distinct() {
    var result = [];

    for (var item in this) {
      if (!result.contains(item)) {
        result.add(item);
      }
    }

    return result;
  }
}

void main() {
  var numbers = [1, 2, 3, 4, 5, 6, 7];
  print(numbers.chunk(3));                // [[1,2,3],[4,5,6],[7]]
  print([1, 2, 2, 3, 3, 3].distinct());   // [1, 2, 3]
  print(numbers.getOrNull(10));
}

Sintaks Generic Dasar

Generic ditulis menggunakan tanda < > untuk menentukan tipe data yang digunakan dalam class atau function. Dengan cara ini, kita bisa menggunakan satu kode yang sama untuk berbagai tipe data tanpa harus membuat ulang.

  • T = Type
  • E = Element (collection)
  • K = Key
  • V = Value
  • R = Return type

Praktik 1 - Generic Class & Fungsi

Repositoryadalah calss generic untuk menyimpan dan mengambil data dengan tipe bebas sesuai T, swap adalah fungsi generic untuk menukar posisi dua data dalam list tanpa tergantung tipe, dan Triple adalah class generic untuk menimpan tiga data dengan tipe yang berbeda dalam satu objek.

DART SOURCE CODE

// Nama  : Marsya Dwi Lintang
// NIM   : 251410028
// Kelas : SI2A

class Repository {
  Map data = {};

  void save(int id, T item) {
    data[id] = item;
  }

  T getById(int id) {
    return data[id]!;
  }

  List getAll() {
    return data.values.toList();
  }
}

void swap(List list, int i, int j) {
  var temp = list[i];
  list[i] = list[j];
  list[j] = temp;
}

class Triple {
  A first;
  B second;
  C third;

  Triple(this.first, this.second, this.third);
}

void main() {
  var repo = Repository();
  repo.save(1, "Data 1");
  repo.save(2, "Data 2");

  print(repo.getById(1));
  print(repo.getAll());

  var list = [1, 2, 3];
  swap(list, 0, 2);
  print(list);

  var t = Triple("Hello", 10, 3.14);
  print("${t.first}, ${t.second}, ${t.third}");
}

Generic vs Dynamic vs Object

  • Generic, Tipe data sudah ditentukan sejak awal, sehingga lebih aman dan tidak perlu konversi saat digunakan.
  • Dynamic, Tipe data bisa berubah-ubah, fleksibel digunakan, tetapi berisiko error saat program berjalan.
  • Object, Bisa menyimpan berbagai tipe data, namun perlu penyesuaian tipe (casting) saat digunakan.
// Contoh perbedaan
Box box1 = Box(42);   // Generic: aman
dynamic box2 = 42;   // Dynamic: tidak aman
Object box3 = 42;   // Object: perlu casting

11. Studi Kasus - Cache Generic

Cache Generic adalah tempat penyimpanan data sementara yang bisa digunakan untuk berbagai tipe data tanpa harus membuat class berbeda-beda.

DART SOURCE CODE

// Nama  : Marsya Dwi Lintang
// NIM   : 251410028
// Kelas : SI2A

class Cache {
  final Map _storage = {};

  void set(String key, T value) {
    _storage[key] = value;
  }

  T get(String key) {
    return _storage[key] as T;
  }

  void remove(String key) {
    _storage.remove(key);
  }
}

class Product {
  String name;
  int price;

  Product(this.name, this.price);
}

void main() {
  var userCache = Cache();
  userCache.set('name', 'Lintang');

  var productCache = Cache();
  productCache.set('p1', Product('Laptop', 1000));

  print(userCache.get('name'));
  print(productCache.get('p1').name);
}

Ringkasan

Generic adalah cara membuat kode yang bisa digunakan untuk berbagai tipe data dengan tetap aman (type-safe), digunakan pada class, function, dan collection agar lebih fleksibel; Dart juga meimilki type inference untuk menebak tipe secara otomatis serta constraints untuk membatasi tipe data sesuai kebutuhan.

Latihan

Latihan ini bertujuan untuk memahami penggunaan generic dalam berbagai bentuk. Class Stack digunakan untuk menyimpan data dengan konsep tumpukan (LIFO), dengan method push, pop, dan peek. Fungsi mergeLists digunakan untuk menggabungkan dua list dengan tipe data yang sama. Sedangkan class Response digunakan sebagai wadah data yang fleksibel, yang bisa menyimpan hasil (data) atau pesan error dengan berbagai tipe.

DART SOURCE CODE

// Nama  : Marsya Dwi Lintang
// NIM   : 251410028
// Kelas : SI2A

class Stack {
  final List _items = [];

  void push(T value) {
    _items.add(value);
  }

  T pop() {
    return _items.removeLast();
  }

  T peek() {
    return _items.last;
  }
}

List mergeLists(List a, List b) {
  return [...a, ...b];
}

class Response {
  T? data;
  String? error;

  Response({this.data, this.error});
}

void main() {
  // Stack
  var stack = Stack();
  stack.push(10);
  stack.push(20);
  print(stack.peek()); // 20
  print(stack.pop());  // 20

  // Merge Lists
  var hasil = mergeLists([1, 2], [3, 4]);
  print(hasil);

  // Response
  var res = Response(data: "Berhasil");
  print(res.data);
}
